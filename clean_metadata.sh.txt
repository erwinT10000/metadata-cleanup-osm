#!/bin/bash

# Script para eliminar metadatos sensibles en im√°genes y PDFs

archivo="$1"

if [[ -z "$archivo" ]]; then
  echo "Uso: $0 archivo"
  exit 1
fi

echo "Limpiando metadatos en: $archivo"

if [[ "$archivo" == *.pdf ]]; then
  pdf-redact-tools --sanitize "$archivo"
elif [[ "$archivo" == *.jpg || "$archivo" == *.jpeg || "$archivo" == *.png ]]; then
  exiftool -all= "$archivo"
else
  echo "Formato no soportado. Solo PDF, JPG, JPEG, PNG."
fi
